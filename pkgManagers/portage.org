  if [[ $(hostname) == "nug.peteches.co.uk" ]]; then
    nproc
  else
    echo $(((($(nproc) + <<distccCpus>>) * 2) + 1))
  fi
#+end_src

#+Name: enabledFeatures
#+begin_src bash :noweb yes
  features=()
  if [[ <<distccCpus>> -gt 0 ]]; then
     features+=("distcc")
  fi
  set -x
  for (( i=0; i<${#features[@]};i++)) ; do
        echo -n ${features[$i]}
        if [[ $i -lt ${#features[@]} ]]; then
          echo -n ' '
        fi
  done
#+end_src
